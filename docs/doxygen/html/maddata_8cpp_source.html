105 bool MadData::fromXml(const QString theXml)
106 {
107  QDomDocument myDocument("mydocument");
108  myDocument.setContent(theXml);
109  QDomElement myTopElement = myDocument.firstChildElement("model");
110  if (myTopElement.isNull())
111  {
112  //TODO - just make this a warning
113  qDebug("the top element couldn't be found!");
114  setGuid(myTopElement.attribute("guid"));
115  mName=MadUtils::xmlDecode(myTopElement.firstChildElement("name").text());
116  mDescription=MadUtils::xmlDecode(myTopElement.firstChildElement("description").text());
117  mImageFile=QString(myTopElement.firstChildElement("imageFile").text());
118  return true;
119  }
120  else
121  return false;
122 }
